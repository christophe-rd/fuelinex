\[
\Delta \beta_1
= N \left(
\beta_1
+ \beta_{S1} S_1
+ \beta_{F1} F_1
+ \beta_{SF1} SF_1, \sigma_{y1}
\right)
\]

\[
\Delta \beta_2
= N \left(
\beta_2
+ \beta_{S2} S_1
+ \beta_{F2} F_1
+ \beta_{SF2} SF_1, \sigma_{y2}
\right)
\]


\textbf{Where:}
\begin{itemize} 
	\item $\beta_1$ = Year 1, treatment Cool Spring, Cool Fall (reference value)
	\item $\beta_1 + \beta_{S1} = $ Year 1, treatment Warm Spring, Cool Fall
	\item $\beta_1 + \beta_{F1} = $ Year 1, treatment Cool Spring, Warm Fall
	\item $\beta_1 + \beta_{S1} + \beta_{F1} + \beta_{SF1} = $ Year 1, treatment Warm Spring, Warm Fall
	\item $ S1 = $ dummy variable, if it's 1, then Warm spring, else Cool Spring
	\item $ F1 = $ dummy variable, if it's 1, then Warm fall, else Cool fall
	\item $ SF1 = $ dummy variable, for a subadditive effect of a Warm Spring and a Warm Fall treatment.
\end{itemize}
